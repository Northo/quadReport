\makeatletter
\def\app@exe{\immediate\write18}
\def\listDir#1{%
  \app@exe{for i in $(ls #1/*); do echo \@backslashchar 1 >> \jobname.tmp; done}%
  \input{\jobname.tmp}%
  \AtEndDocument{\app@exe{rm -f \jobname.tmp}}}
\makeatother
